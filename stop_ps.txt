# Path to PID file
$pidFile = "$env:USERPROFILE\Desktop\HighCPULogger.pid"

# Check if PID file exists
if (Test-Path $pidFile) {
    try {
        $pid = Get-Content $pidFile
        # Try to stop the process
        Stop-Process -Id $pid -Force -ErrorAction Stop
        Write-Host "Stopped process with PID $pid"
    } catch {
        Write-Host "Process with PID $pid not found or already stopped."
    }
    # Always remove the PID file
    Remove-Item $pidFile -Force
    Write-Host "PID file removed."
} else {
    Write-Host "No PID file found. Nothing to stop."
}
